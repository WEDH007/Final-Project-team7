public class book { String title; String genre; String author; int edition; int year; double price; boolean inStock; public book() { title = ""; genre = ""; author = ""; edition = 0; year = 0; price = 0; } public book(String title, String author, String genre, int year, int edition, double price) { this.title = title; this.author = author; this.genre = genre; this.year = year; this.edition = edition; this.price = price; } public String toString(){ return (title + ", " + author + ", " + genre + ", " + year + ", " + edition); } }
